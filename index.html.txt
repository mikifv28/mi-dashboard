function tipoPill(tipo) {
    if (!tipo) return '';
    const cls = tipo.includes('Ases') ? 'badge-a' : tipo.includes('Gest') ? 'badge-g' : 'badge-p';
    const lbl = tipo.includes('Ases') ? 'Asesoramiento' : tipo.includes('Gest') ? 'Gestión' : 'Partícipe';
    return `<span class="badge-tipo ${cls}">${lbl}</span>`;
}

function instrPill(instr) {
    if (!instr) return '';
    let cls = 'pill-mix', lbl = instr;
    if (instr.includes('variable')) { cls = 'pill-rv'; lbl = 'RV'; }
    else if (instr.includes('fija')) { cls = 'pill-rf'; lbl = 'RF'; }
    else if (instr.includes('mixto')) { cls = 'pill-mix'; lbl = 'Mixto'; }
    else if (instr.includes('Commodity') || instr.includes('ETC')) { cls = 'pill-alt'; lbl = 'ETC'; }
    else if (instr.includes('monetario')) { cls = 'pill-mm'; lbl = 'MM'; }
    else if (instr.includes('alternativa')) { cls = 'pill-alt'; lbl = 'Alt'; }
    return `<span class="pill ${cls}">${lbl}</span>`;
}

async function handleFileUpload(event) {
    const file = event.target.files[0];
    if (!file) return;
    
    showLoading();
    
    try {
        const data = await readExcelFile(file);
        RAW_DATA = transformExcelData(data);
        
        // Actualizar fecha
        const now = new Date();
        document.getElementById('fecha-actualizacion').textContent = 
            `Datos actualizados: ${now.toLocaleDateString('es-ES')} ${now.toLocaleTimeString('es-ES')}`;
        
        // Poblar selects
        populateTipoSelect();
        onTipoChange();
        
        hideLoading();
        
    } catch (error) {
        hideLoading();
        alert('Error al procesar el archivo: ' + error.message);
        console.error(error);
    }
}

function showLoading() {
    document.getElementById('loading-overlay').style.display = 'flex';
}

function hideLoading() {
    document.getElementById('loading-overlay').style.display = 'none';
}

async function readExcelFile(file) {
    return new Promise((resolve, reject) => {
        const reader = new FileReader();
        
        reader.onload = (e) => {
            try {
                const data = new Uint8Array(e.target.result);
                const workbook = XLSX.read(data, { type: 'array' });
                
                // Asumimos que la primera hoja contiene los datos
                const firstSheetName = workbook.SheetNames[0];
                const worksheet = workbook.Sheets[firstSheetName];
                
                // Convertir a JSON
                const jsonData = XLSX.utils.sheet_to_json(worksheet, { header: 1 });
                resolve(jsonData);
                
            } catch (error) {
                reject(error);
            }
        };
        
        reader.onerror = (error) => reject(error);
        reader.readAsArrayBuffer(file);
    });
}

function transformExcelData(excelRows) {
    // Esta función debe adaptarse según la estructura de tu Excel
    // Aquí te doy un ejemplo basado en los datos que tenías
    
    const clientes = [];
    
    // Saltamos la primera fila si es cabecera
    for (let i = 1; i < excelRows.length; i++) {
        const row = excelRows[i];
        if (!row || row.length < 5) continue; // Saltar filas vacías
        
        // Adapta estos índices según las columnas de tu Excel
        const cliente = {
            tipo: row[0] || 'A - Asesoramiento',      // Columna A
            grupo: parseInt(row[1]) || 0,              // Columna B
            codigo: row[2]?.toString() || '',          // Columna C
            nombre: row[3] || 'Cliente sin nombre',    // Columna D
            cartera: {
                total_fondos: parseFloat(row[4]) || 0, // Columna E
                rent_esp: parseFloat(row[5]) || 0,     // Columna F
                vol: parseFloat(row[6]) || 0,          // Columna G
                sharpe: parseFloat(row[7]) || 0,       // Columna H
                var95: parseFloat(row[8]) || 0,        // Columna I
                fondos: [],                             // Lo rellenaremos después
                categoria: {},
                divisa: {},
                zona: {},
                sector: {}
            }
        };
        
        clientes.push(cliente);
    }
    
    // Aquí deberías añadir la lógica para procesar los fondos
    // y las distribuciones por categoría, divisa, etc.
    // Basado en cómo tengas estructurado tu Excel
    
    return { clientes };
}

function populateTipoSelect() {
    const tipos = [...new Set(RAW_DATA.clientes.map(c => c.tipo))];
    const selTipo = document.getElementById('sel-tipo');
    selTipo.innerHTML = '<option value="">— Todos los tipos —</option>';
    
    tipos.sort().forEach(tipo => {
        const opt = document.createElement('option');
        opt.value = tipo;
        opt.textContent = tipo;
        selTipo.appendChild(opt);
    });
}

function onTipoChange() {
    const tipo = document.getElementById('sel-tipo').value;
    const selGrupo = document.getElementById('sel-grupo');
    selGrupo.innerHTML = '<option value="">— Todos los grupos —</option>';

    const filtered = RAW_DATA.clientes.filter(c => !tipo || c.tipo === tipo);
    const grupos = [...new Map(filtered.map(c => [String(c.grupo), c])).values()]
        .sort((a, b) => a.grupo - b.grupo);
    
    grupos.forEach(c => {
        const opt = document.createElement('option');
        opt.value = c.grupo;
        opt.textContent = 'Grupo ' + c.grupo;
        selGrupo.appendChild(opt);
    });

    populateClientes();
    const consolidado = consolidarTodos();
    if (consolidado) renderDashboard(consolidado);
}

    if (grupo) {
        const optGroup = document.createElement('option');
        optGroup.value = '__grupo__' + grupo;
        optGroup.textContent = '📊 Ver grupo ' + grupo + ' — consolidado';
        sel.appendChild(optGroup);
    }

    RAW_DATA.clientes.filter(c => {
        if (tipo && c.tipo !== tipo) return false;
        if (grupo && String(c.grupo) !== String(grupo)) return false;
        return true;
    }).forEach(c => {
        const opt = document.createElement('option');
        opt.value = c.codigo;
        opt.textContent = c.nombre + ' (' + c.codigo + ')';
        sel.appendChild(opt);
    });
}

function onClienteChange() {
    const val = document.getElementById('sel-cliente').value;
    
    if (!val) {
        const consolidado = consolidarTodos();
        if (consolidado) renderDashboard(consolidado);
        return;
    }

    if (val.startsWith('__grupo__')) {
        const grupoId = val.replace('__grupo__', '');
        const consolidado = consolidarGrupo(grupoId);
        if (consolidado) renderDashboard(consolidado);
        return;
    }

    const cliente = RAW_DATA.clientes.find(c => String(c.codigo) === String(val));
    if (cliente) renderDashboard(cliente);
}

// FUNCIONES DE CONSOLIDACIÓN (simplificadas)
// ============================================================
function consolidarTodos() {
    const tipo = document.getElementById('sel-tipo').value;
    const grupo = document.getElementById('sel-grupo').value;
    
    const miembros = RAW_DATA.clientes.filter(c =>
        (!tipo || c.tipo === tipo) &&
        (!grupo || String(c.grupo) === String(grupo)) &&
        c.cartera && c.cartera.total_fondos > 0
    );
    
    if (!miembros.length) return null;

    const totalGrupo = miembros.reduce((s, c) => s + (c.cartera.total_fondos || 0), 0);

    // Rentabilidad y volatilidad ponderadas
    const wAvg = (key) => miembros.reduce((s, c) => s + (c.cartera[key] || 0) * (c.cartera.total_fondos / totalGrupo), 0);
    const rent_esp = wAvg('rent_esp');
    const vol = wAvg('vol');
    const sharpe = vol > 0 ? (rent_esp - 3.0) / vol : 0;
    const var95 = vol * 1.645;

    return {
        tipo: tipo || 'Todos',
        codigo: 'TODOS',
        nombre: `Consolidado (${miembros.length} carteras)`,
        _esConsolidado: true,
        _miembros: miembros,
        cartera: {
            total_fondos: totalGrupo,
            rent_esp: Math.round(rent_esp * 100) / 100,
            vol: Math.round(vol * 100) / 100,
            sharpe: Math.round(sharpe * 100) / 100,
            var95: Math.round(var95 * 100) / 100,
            fondos: [],
            categoria: {},
            divisa: {},
            zona: {},
            sector: {}
        }
    };
}

function consolidarGrupo(grupoId) {
    const miembros = RAW_DATA.clientes.filter(c =>
        String(c.grupo) === String(grupoId) &&
        c.cartera && c.cartera.total_fondos > 0
    );
    
    if (!miembros.length) return null;

    const totalGrupo = miembros.reduce((s, c) => s + (c.cartera.total_fondos || 0), 0);
    const wAvg = (key) => miembros.reduce((s, c) => s + (c.cartera[key] || 0) * (c.cartera.total_fondos / totalGrupo), 0);

    return {
        tipo: miembros[0].tipo,
        grupo: grupoId,
        codigo: 'GRUPO-' + grupoId,
        nombre: `Grupo ${grupoId} — Consolidado (${miembros.length} carteras)`,
        _esGrupo: true,
        _miembros: miembros,
        cartera: {
            total_fondos: totalGrupo,
            rent_esp: Math.round(wAvg('rent_esp') * 100) / 100,
            vol: Math.round(wAvg('vol') * 100) / 100,
            sharpe: Math.round((wAvg('rent_esp') - 3.0) / wAvg('vol') * 100) / 100,
            var95: Math.round(wAvg('vol') * 1.645 * 100) / 100,
            fondos: [],
            categoria: {},
            divisa: {},
            zona: {},
            sector: {}
        }
    };
}
